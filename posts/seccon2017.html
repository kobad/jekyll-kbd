Run me

フィボナッチ数列の問題 そのままだと遅くてもとまらない

def fib(n):
    if n <= 1:
        return n
    result = [1, 0, 0, 1]
    matrix = [1, 1, 1, 0]
    while n > 0:
        if n % 2:
            result = mul(matrix, result)
        matrix = mul(matrix, matrix)
        n //= 2
    return result[2]

print(str(fib(11011))[:32])

Putchar music

整形すると

#include <math.h>

int main() {
  int t, i, j;
  unsigned char p[] = "###<f_YM\204g_YM\204g_Y_H #<f_YM\204g_YM\204g_Y_H #+-?[WKAMYJ/7 #+-?[WKgH #+-?[WKAMYJ/7hk\206\203tk\\YJAfkkk";
  for (i = 0, t = 1;; i = (i + 1) % (sizeof(p) - 1)) {
    double x = pow(1.05946309435931, p[i] / 6 + 13);
    for (j = 1 + p[i] % 6; t++ % (8192 / j);)
      putchar(t >> 5 | (int)(t * x));
  }
}

$ gcc music.c
$ ./a.out > a.raw

別ターミナルで

$ sox -V -r 8k -e un -b 8 -c 1 a.raw -b 16 -e si -L a.wav rate -h 44.1k gain -1 trim 0 30

とすると、wavファイルができる

聞くと、スターウォーズの音楽が流れる よって、SECCON{STAR_WARS}がフラグ

SHA-1 is dead

file1 != file2
SHA1(file1) == SHA1(file2)
SHA256(file1) <> SHA256(file2)
2017KiB < sizeof(file1) < 2018KiB
2017KiB < sizeof(file2) < 2018KiB
* 1KiB = 1024 bytes
